258  /**
259   * Creates a new {@link Configurations}.
260   *
261   * <p>The {@link #loadConverters()} and {@link #loadArbiters()}
262   * methods will be invoked during construction.  IF the supplied
263   * {@code configurations} is {@code null}, then the {@link
264   * #loadConfigurations()} method will be invoked during
265   * construction.</p>
266   *
267   * @param configurations a {@link Collection} of {@link
268   * Configuration} instances; if {@code null} then the return value
269   * of the {@link #loadConfigurations()} method will be used instead
270   *
271   * @param converters a {@link Collection} of {@link Converter}
272   * instances; if {@code null} then the return value of the {@link
273   * #loadConverters()} method will be used instead
274   *
275   * @param arbiters a {@link Collection} of {@link Arbiter}
276   * instances; if {@code null} then the return value of the {@link
277   * #loadArbiters()} method will be used instead
278   *
279   * @see #loadConfigurations()
280   *
281   * @see #loadConverters()
282   *
283   * @see #loadArbiters()
284   */
285  public Configurations(Collection<? extends Configuration> configurations,
286                        Collection<? extends Converter<?>> converters,
287                        Collection<? extends Arbiter> arbiters) {
288    super();
289    this.currentlyActiveConfigurations = ThreadLocal.withInitial(() -> new HashSet<>());
290
291    this.expressionFactory = ExpressionFactory.newInstance();
292    assert this.expressionFactory != null;
293    final StandardELContext standardElContext = new StandardELContext(this.expressionFactory);
294    standardElContext.addELResolver(new ConfigurationELResolver());
295    this.elContext = standardElContext;
296    
297    if (configurations == null) {
298      configurations = this.loadConfigurations();
299    }
300    if (configurations == null || configurations.isEmpty()) {
301      this.configurations = Collections.emptySet();
302    } else {
303      this.configurations = Collections.unmodifiableCollection(new LinkedList<>(configurations));
304    }
305    for (final Configuration configuration : configurations) {
306      if (configuration != null) {
307        configuration.setConfigurations(this);
308      }
309    }
310
311    if (converters == null) {
312      converters = this.loadConverters();
313    }
314    if (converters == null || converters.isEmpty()) {
315      converters = Collections.emptySet();
316    } else {
317      converters = Collections.unmodifiableCollection(new LinkedList<>(converters));
318    }
319    assert converters != null;
320    this.converters = Collections.unmodifiableMap(converters.stream().collect(Collectors.toMap(c -> c.getType(), Function.identity())));
321
322    if (arbiters == null) {
323      arbiters = this.loadArbiters();
324    }
325    if (arbiters == null || arbiters.isEmpty()) {
326      this.arbiters = Collections.emptySet();
327    } else {
328      this.arbiters = Collections.unmodifiableCollection(new LinkedList<>(arbiters));
329    }
330
331    this.initialized = true;
332    
333    final Map<String, String> coordinates = this.getValue(null, CONFIGURATION_COORDINATES, new TypeLiteral<Map<String, String>>() {
334        private static final long serialVersionUID = 1L; }.getType());
335    if (coordinates == null || coordinates.isEmpty()) {
336      this.configurationCoordinates = Collections.emptyMap();
337    } else {
338      this.configurationCoordinates = Collections.unmodifiableMap(coordinates);
339    }
340
341  }

349  /**
350   * Loads a {@link Collection} of {@link Configuration} objects and
351   * returns it.
352   *
353   * <p>This method never returns {@code null}.</p>
354   *
355   * <p>Overrides of this method must not return {@code null}.</p>
356   *
357   * <p>The default implementation of this method uses the {@link
358   * ServiceLoader} mechanism to load {@link Configuration}
359   * instances.</p>
360   *
361   * @return a non-{@code null}, {@link Collection} of {@link
362   * Configuration} instances
363   *
364   * @see ServiceLoader#load(Class)
365   */
366  protected Collection<? extends Configuration> loadConfigurations() {
367    final String cn = this.getClass().getName();
368    final String mn = "loadConfigurations";
369    if (this.logger.isLoggable(Level.FINER)) {
370      this.logger.entering(cn, mn);
371    }
372    final Collection<Configuration> returnValue = new LinkedList<>();
373    final ServiceLoader<Configuration> configurationLoader = ServiceLoader.load(Configuration.class);
374    assert configurationLoader != null;
375    final Iterator<Configuration> configurationIterator = configurationLoader.iterator();
376    assert configurationIterator != null;
377    while (configurationIterator.hasNext()) {
378      final Configuration configuration = configurationIterator.next();
379      assert configuration != null;
380      returnValue.add(configuration);
381    }
382    if (this.logger.isLoggable(Level.FINER)) {
383      this.logger.exiting(cn, mn, returnValue);
384    }
385    return returnValue;
386  }

509  /**
510   * Returns a configuration value corresponding to the configuration
511   * property suitable for the supplied {@code
512   * configurationCoordinates} and {@code name}, or the supplied
513   * {@code defaultValue} if {@code null} would otherwise be returned,
514   * converted, if possible, to the type represented by the supplied
515   * {@code type}.
516   *
517   * <p>This method may return {@code null}.</p>
518   *
519   * @param <T> the type to which a {@link String}-typed configuration
520   * value should be converted
521   *
522   * @param configurationCoordinates a {@link Map} representing the
523   * configuration coordinates in effect for this request; may be
524   * {@code null}
525   *
526   * @param name the name of the configuration property for which a
527   * value will be returned; must not be {@code null}
528   *
529   * @param type a {@link Type} representing the type to
530   * which the configuration value will be {@linkplain
531   * Converter#convert(String) converted}; must not be {@code null}
532   *
533   * @param defaultValue the value that will be converted if {@code
534   * null} would otherwise be returned; may be {@code null}
535   *
536   * @return the configuration value, or {@code null}
537   *
538   * @exception NullPointerException if {@code name} or {@code type}
539   * is {@code null}
540   *
541   * @exception NoSuchConverterException if there is no {@link
542   * Converter} available that {@linkplain Converter#getType()
543   * handles} the {@link Type} represented by the supplied {@code
544   * type}
545   *
546   * @exception ConversionException if type conversion could not occur
547   * for any reason
548   *
549   * @exception AmbiguousConfigurationValuesException if two or more
550   * values were found that could be suitable and arbitration
551   * {@linkplain #performArbitration(Map, String, Collection) was
552   * performed} but could not resolve the dispute
553   *
554   * @see #getValue(Map, String, Converter, String)
555   */
556  @Override
557  public final <T> T getValue(final Map<String, String> configurationCoordinates, final String name, Type type, final String defaultValue) {
558    final String cn = this.getClass().getName();
559    final String mn = "getValue";
560    if (this.logger.isLoggable(Level.FINER)) {
561      this.logger.entering(cn, mn, new Object[] { configurationCoordinates, name, type, defaultValue });
562    }
563    if (type instanceof Class) {
564      final Class<?> c = (Class<?>)type;
565      if (c.isPrimitive()) {
566        type = wrapperTypes.get(c);
567      }
568    }
569    @SuppressWarnings("unchecked")
570    final Converter<T> converter = (Converter<T>)this.converters.get(type);
571    if (converter == null) {
572      throw new NoSuchConverterException(type);
573    }
574    if (this.logger.isLoggable(Level.FINE)) {
575      this.logger.logp(Level.FINE, cn, mn, "Using {0} to convert String to {1}", new Object[] { converter, type });
576    }
577    final T returnValue = this.getValue(configurationCoordinates, name, converter, defaultValue);
578    if (this.logger.isLoggable(Level.FINER)) {
579      this.logger.exiting(cn, mn, returnValue);
580    }
581    return returnValue;
582  }

664   * Returns an object that is the value for the configuration request
665   * represented by the supplied {@code configurationCoordinates},
666   * {@code name} and {@code defaultValue} parameters, as converted by
667   * the supplied {@link Converter}.
668   *
669   * <p>This method may return {@code null}.</p>
670   *
671   * @param <T> the type of the object to be returned
672   *
673   * @param configurationCoordinates the configuration coordinates for which
674   * a value should be selected; may be {@code null}
675   *
676   * @param name the name of the configuration property within the
677   * world defined by the supplied {@code configurationCoordinates}
678   * whose value is to be selected; must not be {@code null}
679   *
680   * @param converter a {@link Converter} instance that will convert
681   * any {@link String} configuration value into the type of object
682   * that this method will return; must not be {@code null}
683   *
684   * @param defaultValue the fallback default value to use as an
685   * absolute last resort; may be {@code null}; will also be converted
686   * by the supplied {@link Converter}
687   *
688   * @return the value for the implied configuration property, or {@code null}
689   *
690   * @exception NullPointerException if either {@code name} or {@code
691   * converter} is {@code null}
692   *
693   * @exception ConversionException if type conversion could not occur
694   * for any reason
695   *
696   * @exception AmbiguousConfigurationValuesException if two or more
697   * values were found that could be suitable and arbitration
698   * {@linkplain #performArbitration(Map, String, Collection) was
699   * performed} but could not resolve the dispute
700   *
701   * @see Converter#convert(String)
702   *
703   * @see Configuration#getValue(Map, String)
704   *
705   * @see #performArbitration(Map, String, Collection)
706   *
707   * @see #handleMalformedConfigurationValues(Collection)
708   */
709  public <T> T getValue(Map<String, String> configurationCoordinates, final String name, final Converter<T> converter, final String defaultValue) {
710    final String cn = this.getClass().getName();
711    final String mn = "getValue";
712    if (this.logger.isLoggable(Level.FINER)) {
713      this.logger.entering(cn, mn, new Object[] { configurationCoordinates, name, converter, defaultValue });
714    }
715    Objects.requireNonNull(name);
716    Objects.requireNonNull(converter);
717    this.checkState();
718    if (configurationCoordinates == null) {
719      configurationCoordinates = Collections.emptyMap();
720    }
721    T returnValue = null;
722
723    ConfigurationValue selectedValue = null;    
724
725    // We use a PriorityQueue of ConfigurationValues sorted by their
726    // specificity (most specific first) to keep track of the most
727    // specific ConfigurationValue found so far.  We create it only
728    // when necessary.
729    final Comparator<ConfigurationValue> comparator = Comparator.<ConfigurationValue>comparingInt(v -> v.specificity()).reversed();
730    PriorityQueue<ConfigurationValue> values = null;
731
732    Collection<ConfigurationValue> badValues = null;
733    
734    for (final Configuration configuration : this.configurations) {
735      assert configuration != null;
736
737      final ConfigurationValue value;
738      try {
739        if (isActive(configuration)) {
740          value = null;
741        } else {
742          this.activate(configuration);
743          value = configuration.getValue(configurationCoordinates, name);
744        }
745      } finally {
746        this.deactivate(configuration);
747      }
748
749      if (value != null) {
750
751        if (!name.equals(value.getName())) {
752          badValues.add(value);
753          continue;
754        }
755
756        Map<String, String> valueCoordinates = value.getCoordinates();
757        if (valueCoordinates == null) {
758          valueCoordinates = Collections.emptyMap();
759        }
760
761        final int configurationCoordinatesSize = configurationCoordinates.size();
762        final int valueCoordinatesSize = valueCoordinates.size();
763
764        if (configurationCoordinatesSize < valueCoordinatesSize) {
765          // Bad value!
766          if (badValues == null) {
767            badValues = new LinkedList<>();
768          }
769          badValues.add(value);
770          
771        } else if (configurationCoordinates.equals(valueCoordinates)) {
772          // We have an exact match.  We hope it's going to be the
773          // only one.
774          
775          if (selectedValue == null) {
776            
777            if (values == null || values.isEmpty()) {
778              // There aren't any conflicts yet; this is good.  This
779              // value will be our candidate.
780              selectedValue = value;
781
782            } else {
783              // We got a match, but we already *had* a match, so we
784              // don't have a candidate--instead, add it to the bucket
785              // of values that will be arbitrated later.
786              values.add(value);
787              
788            }
789            
790          } else {
791            assert selectedValue != null;
792            // We have an exact match, but we already identified a
793            // candidate, so oops, we have to treat our prior match
794            // and this one as non-candidates.
795            
796            if (values == null) {
797              values = new PriorityQueue<>(comparator);
798            }
799            values.add(selectedValue);
800            selectedValue = null;
801            values.add(value);
802          }
803
804        } else if (configurationCoordinatesSize == valueCoordinatesSize) {
805          // Bad value!  The configuration subsystem handed back a
806          // value containing coordinates not drawn from the
807          // configurationCoordinatesSet.  We know this because we already
808          // tested for Set equality, which failed, so this test means
809          // disparate entries.
810          if (badValues == null) {
811            badValues = new LinkedList<>();
812          }
813          badValues.add(value);
814          
815        } else if (selectedValue != null) {
816          // Nothing to do; we've already got our candidate.  We don't
817          // break here because we're going to ensure there aren't any
818          // duplicates.
819          
820        } else if (configurationCoordinates.entrySet().containsAll(valueCoordinates.entrySet())) {
821          // We specified, e.g., {a=b, c=d, e=f} and they have, say,
822          // {c=d, e=f} or {a=b, c=d} etc. but not, say, {q=r}.
823          if (values == null) {
824            values = new PriorityQueue<>(comparator);
825          }
826          values.add(value);
827          
828        } else {
829          // Bad value!
830          if (badValues == null) {
831            badValues = new LinkedList<>();
832          }
833          badValues.add(value);
834          
835        }
836      }
837    }
838    assert this.allConfigurationsInactive();
839
840    if (badValues != null && !badValues.isEmpty()) {
841      this.handleMalformedConfigurationValues(badValues);
842    }
843    
844    if (selectedValue == null) {
845      final Collection<ConfigurationValue> valuesToArbitrate = new LinkedList<>();
846      int highestSpecificitySoFarEncountered = -1;
847      if (values != null) {
848        while (!values.isEmpty()) {
849          
850          final ConfigurationValue value = values.poll();
851          assert value != null;
852          
853          // The values are sorted by their specificity, most specific
854          // first.
855          final int valueSpecificity = Math.max(0, value.specificity());
856          assert highestSpecificitySoFarEncountered < 0 || valueSpecificity <= highestSpecificitySoFarEncountered;
857          
858          if (highestSpecificitySoFarEncountered < 0 || valueSpecificity < highestSpecificitySoFarEncountered) {
859            if (selectedValue == null) {
860              assert valuesToArbitrate.isEmpty();
861              selectedValue = value;
862              highestSpecificitySoFarEncountered = valueSpecificity;
863            } else if (valuesToArbitrate.isEmpty()) {
864              break;
865            } else {
866              valuesToArbitrate.add(value);
867            }
868          } else if (valueSpecificity == highestSpecificitySoFarEncountered) {
869            assert selectedValue != null;
870            if (value.isAuthoritative()) {
871              if (selectedValue.isAuthoritative()) {
872                // Both say they're authoritative; arbitration required
873                valuesToArbitrate.add(selectedValue);
874                selectedValue = null;
875                valuesToArbitrate.add(value);
876              } else {
877                // value is authoritative; selectedValue is not; so swap
878                // them
879                selectedValue = value;
880              }
881            } else if (selectedValue.isAuthoritative()) {
882              // value is not authoritative; selected value is; so just
883              // drop value on the floor; it's not authoritative.
884            } else {
885              // Neither is authoritative; arbitration required.
886              valuesToArbitrate.add(selectedValue);
887              selectedValue = null;
888              valuesToArbitrate.add(value);
889            }
890          } else {
891            assert false : "valueSpecificity > highestSpecificitySoFarEncountered: " + valueSpecificity + " > " + highestSpecificitySoFarEncountered;
892          }
893          
894        }
895      }
896
897      if (selectedValue == null) {
898        selectedValue = this.performArbitration(configurationCoordinates, name, Collections.unmodifiableCollection(valuesToArbitrate));
899      }
900    }
901
902    if (selectedValue == null) {
903      if (defaultValue == null) {
904        returnValue = converter.convert(null);
905      } else {
906        returnValue = converter.convert(this.interpolate(defaultValue));
907      }
908    } else {
909      final String valueToConvert = selectedValue.getValue();
910      if (valueToConvert == null) {
911        returnValue = converter.convert(null);
912      } else {
913        returnValue = converter.convert(this.interpolate(valueToConvert));
914      }
915    }
916
917    if (this.logger.isLoggable(Level.FINER)) {
918      this.logger.exiting(cn, mn, returnValue);
919    }
920    return returnValue;
921  }

923  /**
924   * Interpolates any expressions occurring within the supplied {@code
925   * value} and returns the result of interpolation.
926   *
927   * <p>This method may return {@code null}.</p>
928   *
929   * <p>Overrides of this method may return {@code null}.</p>
930   *
931   * <p>The default implementation of this method performs
932   * interpolation by using a {@link ValueExpression} as {@linkplain
933   * ExpressionFactory#createValueExpression(ELContext, String, Class)
934   * produced by an <code>ExpressionFactory</code>}.</p>
935   *
936   * <p>A {@code configurations} object is made available to any
937   * Expression Language expressions, which exposes this {@link
938   * Configurations} object.  This means, among other things, that you
939   * can retrieve configuration values using the following syntax:</p>
940   * <pre>${configurations["java.home"]}</pre>
941   *
942   * @param value a configuration value {@link String}, before any
943   * type conversion has taken place, with (possibly) expression
944   * language expressions in it; may be {@code null} in which case
945   * {@code null} is returned
946   *
947   * @return the result of interpolating the supplied {@code value},
948   * or {@code null}
949   *
950   * @exception PropertyNotFoundException if the supplied {@code
951   * value} contained a valid expression language expression that
952   * identifies an unknown property
953   *
954   * @see <a
955   * href="https://docs.oracle.com/javaee/7/tutorial/jsf-el.htm#GJDDD">the
956   * section in the Java EE Tutorial on the Unified Expression
957   * Language</a>
958   */
959  public String interpolate(final String value) {
960    final String cn = this.getClass().getName();
961    final String mn = "interpolate";
962    if (this.logger.isLoggable(Level.FINER)) {
963      this.logger.entering(cn, mn, value);
964    }
965    final String returnValue;
966    if (value == null) {
967      returnValue = null;
968    } else {
969      final ValueExpression valueExpression = this.expressionFactory.createValueExpression(this.elContext, value, String.class);
970      assert valueExpression != null;
971      returnValue = String.class.cast(valueExpression.getValue(this.elContext));
972    }
973    if (this.logger.isLoggable(Level.FINER)) {
974      this.logger.exiting(cn, mn, returnValue);
975    }
976    return returnValue;
977  }

979  /**
980   * Returns a {@link Set} of names of {@link ConfigurationValue}s
981   * that might be returned by this {@link Configurations} instance.
982   *
983   * <p>This method does not return {@code null}.</p>
984   *
985   * <p>Overrides of this method must not return {@code null}.</p>
986   *
987   * <p>Just because a name appears in the returned {@link Set} does
988   * <em>not</em> mean that a {@link ConfigurationValue} <em>will</em>
989   * be returned for it in a location in configuration space
990   * identified by any arbitrary set of configuration coordinates.</p>
991   *
992   * @return a non-{@code null} {@link Set} of names of {@link
993   * ConfigurationValue}s
994   */
995  @Override
996  public Set<String> getNames() {
997    final Set<String> returnValue;
998    if (this.configurations == null || this.configurations.isEmpty()) {
999      returnValue = Collections.emptySet();
1000    } else {
1001      final Set<String> names = new TreeSet<>();
1002      for (final Configuration configuration : this.configurations) {
1003        if (configuration != null) {
1004          final Set<String> configurationNames = configuration.getNames();
1005          if (configurationNames != null && !configurationNames.isEmpty()) {
1006            names.addAll(configurationNames);
1007          }
1008        }
1009      }
1010      if (names.isEmpty()) {
1011        returnValue = Collections.emptySet();
1012      } else {
1013        returnValue = Collections.unmodifiableSet(names);
1014      }
1015    }
1016    return returnValue;
1017  }

1035  /**
1036   * Given a logical request for a configuration value, represented by
1037   * the {@code configurationCoordinates} and {@code name} parameter
1038   * values, and a {@link Collection} of {@link ConfigurationValue}
1039   * instances that represents the ambiguous response from several
1040   * {@link Configuration} instances, attempts to resolve the
1041   * ambiguity by returning a single {@link ConfigurationValue}
1042   * instead.
1043   *
1044   * <p>This method may return {@code null}.</p>
1045   *
1046   * <p>Overrides of this method may return {@code null}.</p>
1047   *
1048   * <p>The default implementation of this method asks all registered
1049   * {@link Arbiter}s in turn to perform the arbitration and returns
1050   * the first non-{@code null} response received.</p>
1051   *
1052   * @param configurationCoordinates the ({@linkplain
1053   * Collections#unmodifiableMap(Map) immutable}) configuration
1054   * coordinates in effect for the request; may be {@code null}
1055   *
1056   * @param name the name of the configuration value; may be {@code
1057   * null}
1058   *
1059   * @param values an {@linkplain
1060   * Collections#unmodifiableCollection(Collection) immutable} {@link
1061   * Collection} of definitionally ambiguous {@link
1062   * ConfigurationValue}s that resulted from the request; may be
1063   * {@code null}
1064   *
1065   * @return the result of arbitration, or {@code null}
1066   *
1067   * @exception AmbiguousConfigurationValuesException if successful
1068   * arbitration did not happen for any reason
1069   *
1070   * @see Arbiter
1071   */
1072  protected ConfigurationValue performArbitration(final Map<? extends String, ? extends String> configurationCoordinates,
1073                                                  final String name,
1074                                                  final Collection<? extends ConfigurationValue> values) {
1075    final String cn = this.getClass().getName();
1076    final String mn = "performArbitration";
1077    if (this.logger.isLoggable(Level.FINER)) {
1078      this.logger.entering(cn, mn, new Object[] { configurationCoordinates, name, values });
1079    }
1080
1081    ConfigurationValue returnValue = null;
1082    if (this.arbiters != null && !this.arbiters.isEmpty()) {
1083      for (final Arbiter arbiter : arbiters) {
1084        if (arbiter != null) {
1085          final ConfigurationValue arbitrationResult = arbiter.arbitrate(configurationCoordinates, name, values);
1086          if (arbitrationResult != null) {
1087            returnValue = arbitrationResult;
1088            break;
1089          }
1090        }
1091      }
1092    }
1093    if (returnValue == null && values != null && !values.isEmpty()) {
1094      throw new AmbiguousConfigurationValuesException(null, null, configurationCoordinates, name, values);
1095    }
1096    
1097    if (this.logger.isLoggable(Level.FINER)) {
1098      this.logger.exiting(cn, mn, returnValue);
1099    }
1100    return returnValue;
1101  }

1369     * {@code base} object with it.  If the return value of that
1370     * method invocation is {@code null}, then a {@link
1371     * PropertyNotFoundException} is thrown; otherwise the return
1372     * value of its {@link ConfigurationValue#getValue()} method is
1373     * returned.</p>
1374     *
1375     * <p>This method returns {@code null} in all other cases.</p>
1376     *
1377     * @param elContext the {@link ELContext} in effect; must not be
1378     * {@code null}
1379     *
1380     * @param base the base; may be {@code null}
1381     *
1382     * @param property the property; may be {@code null}
1383     *
1384     * @return a {@link Configurations} object, a {@link String} or
1385     * {@code null}
1386     *
1387     * @exception NullPointerException if {@code elContext} is {@code
1388     * null}
1389     *
1390     * @exception PropertyNotFoundException if {@code property} is a
1391     * {@link String} but does not identify a configuration property
1392     * that has a value
1393     *
1394     * @see ELResolver#getValue(ELContext, Object, Object)
1395     *
1396     * @see Configurations#getValue(String)
1397     */
1398    @Override
1399    public final Object getValue(final ELContext elContext, final Object base, final Object property) {
1400      Objects.requireNonNull(elContext);
1401      Object returnValue = null;
1402      if (base == null) {
1403        if ("configurations".equals(property)) {
1404          elContext.setPropertyResolved(true);
1405          returnValue = Configurations.this;
1406        }
1407      } else if (base instanceof Configurations) {
1408        if (property instanceof String) {
1409          final String configurationPropertyName = (String)property;
1410          returnValue = ((Configurations)base).getValue(configurationPropertyName);
1411          elContext.setPropertyResolved(true);
1412          if (returnValue == null) {
1413            throw new PropertyNotFoundException(configurationPropertyName);
1414          }          
1415        }
1416      }
1417      return returnValue;
1418    }

1421     * Effectively does nothing when invoked.
1422     *
1423     * @param elContext the {@link ELContext} in effect; ignored; may be {@code null}
1424     *
1425     * @param base the base; ignored; may be {@code null}
1426     *
1427     * @param property the property; ignored; may be {@code null}
1428     *
1429     * @param value the value; ignored; may be {@code null}

1432     *
1433     * @see #isReadOnly(ELContext, Object, Object)
1434     */
1435    @Override
1436    public final void setValue(final ELContext elContext, final Object base, final Object property, final Object value) {
1437      if (elContext != null) {
1438        elContext.setPropertyResolved(false);
1439      }
1440    }
