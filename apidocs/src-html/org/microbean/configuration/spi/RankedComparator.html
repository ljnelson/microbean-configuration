017package org.microbean.configuration.spi;
018
019import java.io.Serializable;
020
021import java.util.ArrayList;
022import java.util.Collections;
023import java.util.Comparator;
024import java.util.List;
025import java.util.Objects;
026
027/**
028 * A {@link Comparator} that uses a {@link List} of {@link Object}s to
029 * rank {@linkplain #getComparisonObject(Object) other objects} for
030 * comparison purposes.
031 *
032 * @param <T> the type of object to {@linkplain #compare(Object,
033 * Object) compare}
034 *
035 * @author <a href="https://about.me/lairdnelson"
036 * target="_parent">Laird Nelson</a>
037 *
038 * @see #RankedComparator(List)
039 *
040 * @see #compare(Object, Object)
041 */
042public class RankedComparator<T> implements Comparator<T>, Serializable {
043
044
045  /*
046   * Static fields.
047   */
048
049
050  /**
051   * The version of this class for {@linkplain Serializable
052   * serialization purposes}.
053   */
054  private static final long serialVersionUID = 1L;
055
056
057  /*
058   * Instance fields.
059   */
060
061
062  /**
063   * An {@linkplain Collections#unmodifiableList(List) unmodifiable}
064   * {@link List} of {@link Object}s to be used for ranking purposes;
065   * the {@link Object} at position {@code 0} is deemed to have the
066   * highest rank.
067   *
068   * <p>This field is never {@code null}.</p>
069   *
070   * @see #RankedComparator(List)
071   */
072  private final List<?> items;
073
074
075  /*
076   * Constructors.
077   */
078
079
080  /**
081   * Creates a new {@link RankedComparator}.
082   *
083   * @param items a {@link List} of {@link Object}s that will be used
084   * for ranking purposes; the item at position {@code 0} is deemed to
085   * outrank all others; copied by value
086   *
087   * @see #getComparisonObject(Object)
088   */
089  public RankedComparator(final List<?> items) {
090    super();
091    if (items == null || items.isEmpty()) {
092      this.items = Collections.emptyList();
093    } else {
094      this.items = Collections.unmodifiableList(new ArrayList<>(items));
095    }
096  }
097
098
099  /*
100   * Instance methods.
101   */
102
103
104  /**
105   * Returns {@code true} if this {@link RankedComparator} explicitly
106   * ranks the supplied object.
107   *
108   * @param object the object to test; may be {@code null}
109   *
110   * @return {@code true} if this {@link RankedComparator} explicitly
111   * ranks the supplied object; {@code false} otherwise
112   *
113   * @see #getComparisonObject(Object)
114   */
115  public final boolean ranks(final T object) {
116    final Object comparisonObject = this.getComparisonObject(object);
117    return comparisonObject != null && this.items.contains(comparisonObject);
118  }
119
120  /**
121   * Compares two objects, returning {@code -1} if {@code one}
122   * outranks {@code two}, {@code 1} if {@code two} outranks {@code
123   * one}, and {@code 0} if the two objects are either ranked equally
124   * or their ranks could not be determined.
125   *
126   * <p>This method makes use of the {@link List} {@linkplain
127   * #RankedComparator(List) supplied at construction time} for
128   * ranking information.</p>
129   *
130   * <p>Non-{@code null} objects outrank {@code null} objects.</p>
131   *
132   * @param one the first object; may be {@code null}
133   *
134   * @param two the second object; may be {@code null}
135   *
136   * @return {@code -1} if {@code one} outranks {@code two}, {@code 1}
137   * if {@code two} outranks {@code one}, and {@code 0} if the two
138   * objects are either ranked equally or their ranks could not be
139   * determined
140   *
141   * @see #getComparisonObject(Object)
142   *
143   * @see #RankedComparator(List)
144   */
145  @Override
146  public final int compare(final T one, final T two) {
147    final int returnValue;
148    if (one == two) {
149      returnValue = 0;
150    } else if (one == null) {
151      assert two != null;
152      returnValue = 1; // nulls sort "right"; non-nulls win
153    } else if (two == null) {
154      assert one != null;
155      returnValue = -1; // nulls sort "right"; non-nulls win
156    } else {
157      final Object oneComparisonObject = this.getComparisonObject(one);
158      final int oneIndex = oneComparisonObject == null ? -1 : this.items.indexOf(oneComparisonObject);
159      final Object twoComparisonObject = this.getComparisonObject(two);
160      final int twoIndex = twoComparisonObject == null ? -1 : this.items.indexOf(twoComparisonObject);
161      if (oneIndex < 0) {
162        if (twoIndex < 0) {
163          returnValue = 0;
164        } else {
165          returnValue = 1; // two "won"; it's ranked; one isn't
166        }
167      } else if (twoIndex < 0) {
168        returnValue = -1; // one "won"; it's ranked; two isn't
169      } else if (oneIndex > twoIndex) {
170        returnValue = 1; // two "won"; its rank is closer to 0
171      } else if (oneIndex == twoIndex) {
172        returnValue = 0;
173      } else {
174        returnValue = -1; // one "won"; its rank is closer to 0
175      }
176    }
177    return returnValue;
178  }
179
180  /**
181   * Given an object to be {@linkplain #compare(Object, Object)
182   * compared}, returns the {@link Object} that should be used to
183   * determine ranking (as determined by the {@link List} {@linkplain
184   * #RankedComparator(List) supplied at construction time}).
185   *
186   * <p>This method may return {@code null}.</p>
187   *
188   * <p>Overrides of this method may return {@code null}.</p>
189   *
190   * <p>The default implementation of this method simply returns the
191   * supplied {@code realObject} parameter value.</p>
192   *
193   * @param realObject the object to be {@linkplain #compare(Object,
194   * Object) compared}; may be {@code null}
195   *
196   * @return the {@link Object} that should be used to determine
197   * ranking (as determined by the {@link List} {@linkplain
198   * #RankedComparator(List) supplied at construction time}), possibly
199   * {@code null}
200   *
201   * @see #RankedComparator(List)
202   *
203   * @see #compare(Object, Object)
204   */
205  protected Object getComparisonObject(final T realObject) {
206    return realObject;
207  }
208  
209}
