189  public static final class PropertiesConfiguration extends AbstractConfiguration implements Serializable {
190
191    private static final long serialVersionUID = 1L;
192    
193    private final Map<String, String> coordinates;
194    
195    private final Properties properties;
196    
197    public PropertiesConfiguration(final Map<String, String> coordinates, final Properties properties) {
198      super();
199      this.coordinates = coordinates == null ? Collections.emptyMap() : coordinates;
200      this.properties = properties == null ? new Properties() : properties;
201    }
202
203    public Properties getProperties() {
204      return this.properties;
205    }
206
207    @Override
208    public ConfigurationValue getValue(final Map<String, String> applicationCoordinates, final String name) {
209      final Properties properties = this.getProperties();
210      assert properties != null;
211      final Set<String> stringPropertyNames = properties.stringPropertyNames();
212      assert stringPropertyNames != null;
213      if (stringPropertyNames.contains(name)) {
214        return new ConfigurationValue(this, this.coordinates, name, properties.getProperty(name), false);
215      } else {
216        return null;
217      }
218    }
219    
220  }
